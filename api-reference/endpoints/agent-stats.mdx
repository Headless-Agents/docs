---
title: 'Agent Statistics'
api: 'GET /agent/{agent_id}/stats'
description: 'Retrieve usage statistics for a specific agent'
---

Get detailed usage statistics for a specific agent, including message distribution, response times, and usage timeline.

## Path Parameters

<ParamField path="agent_id" type="string" required>
  ID of the agent to get statistics for
</ParamField>

## Response

<ResponseField name="message_distribution" type="object" required>
  Statistics about message distribution
  <Expandable>
    <ResponseField name="sample_size" type="integer">Number of messages in the sample</ResponseField>
    <ResponseField name="assistant_messages" type="integer">Number of assistant messages</ResponseField>
    <ResponseField name="user_messages" type="integer">Number of user messages</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="response_time" type="object" required>
  Statistics about response times
  <Expandable>
    <ResponseField name="sample_size" type="integer">Number of responses in the sample</ResponseField>
    <ResponseField name="min_seconds" type="number">Minimum response time in seconds</ResponseField>
    <ResponseField name="average_seconds" type="number">Average response time in seconds</ResponseField>
    <ResponseField name="max_seconds" type="number">Maximum response time in seconds</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="message_length" type="object" required>
  Statistics about message lengths
  <Expandable>
    <ResponseField name="user" type="object">
      User message length statistics
      <Expandable>
        <ResponseField name="sample_size" type="integer">Number of messages in sample</ResponseField>
        <ResponseField name="average" type="number">Average message length</ResponseField>
        <ResponseField name="min" type="integer">Minimum message length</ResponseField>
        <ResponseField name="max" type="integer">Maximum message length</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="assistant" type="object">
      Assistant message length statistics
      <Expandable>
        <ResponseField name="sample_size" type="integer">Number of messages in sample</ResponseField>
        <ResponseField name="average" type="number">Average message length</ResponseField>
        <ResponseField name="min" type="integer">Minimum message length</ResponseField>
        <ResponseField name="max" type="integer">Maximum message length</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="usage_timeline" type="object" required>
  Timeline of agent usage
  <Expandable>
    <ResponseField name="last_24h" type="integer">Usage in last 24 hours</ResponseField>
    <ResponseField name="last_7d" type="integer">Usage in last 7 days</ResponseField>
    <ResponseField name="last_30d" type="integer">Usage in last 30 days</ResponseField>
    <ResponseField name="first_conversation" type="string">Timestamp of first conversation</ResponseField>
    <ResponseField name="last_conversation" type="string">Timestamp of last conversation</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total_conversations" type="integer" required>
  Total number of conversations
</ResponseField>

<ResponseField name="total_messages" type="integer" required>
  Total number of messages
</ResponseField>

<ResponseField name="stats_generated_at" type="string" required>
  Timestamp when these statistics were generated
</ResponseField>
